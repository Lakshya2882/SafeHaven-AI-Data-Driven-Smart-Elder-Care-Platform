# Extract the starting age to help group later
def extract_start_age(age_str):
    if "to" in age_str:
        return int(age_str.split(" to ")[0])
    elif "85" in age_str or "100" in age_str:
        return 85
    else:
        return np.nan

df["Start_Age"] = df["Age Group"].apply(extract_start_age)

# Assign custom age group labels
def group_age(age):
    if pd.isna(age):  # For "All ages"
        return "All ages"
    elif age <= 14:
        return "0-14"
    elif 15 <= age <= 24:
        return "15-24"
    elif 25 <= age <= 54:
        return "25-54"
    elif 55 <= age <= 64:
        return "55-64"
    else:
        return "65+"

df["Age_Range"] = df["Start_Age"].apply(group_age)

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
from statsmodels.tsa.statespace.sarimax import SARIMAX
import warnings
warnings.filterwarnings("ignore")


age_groups = df[df["Age_Range"] != "All ages"]["Age_Range"].tolist()
years = df.columns[1:-1].astype(int)

# Plot setup
plt.figure(figsize=(14, 8))

for age_group in age_groups:
    values = df[df["Age_Range"] == age_group].iloc[0, 1:-1].astype(float).values
    y = pd.Series(values, index=pd.date_range(start=f'{years[0]}', periods=len(years), freq='Y'))

    # Fit SARIMA model (adjust seasonal_order as needed)
    model = SARIMAX(y, order=(1, 1, 1), seasonal_order=(1, 1, 0, 5))
    results = model.fit()

    # Forecast
    steps = 10
    pred_uc = results.get_forecast(steps=steps)
    pred_ci = pred_uc.conf_int()
    pred_index = pd.date_range(start=y.index[-1] + pd.offsets.YearEnd(1), periods=steps, freq='Y')
    pred_mean = pd.Series(pred_uc.predicted_mean.values, index=pred_index)
    lower = pd.Series(pred_ci.iloc[:, 0].values, index=pred_index)
    upper = pd.Series(pred_ci.iloc[:, 1].values, index=pred_index)

    # Plotting
    plt.plot(y.index, y.values, label=f"{age_group} (Observed)", linewidth=1.5)
    plt.plot(pred_index, pred_mean, linestyle='--', label=f"{age_group} (Forecast)")
    plt.fill_between(pred_index, lower, upper, alpha=0.2)

# Final touches
plt.title("SARIMA Forecast of Population by Age Group")
plt.xlabel("Year")
plt.ylabel("Population")
plt.legend(loc='upper left', fontsize=9)
plt.grid(True)
plt.tight_layout()
plt.show()
