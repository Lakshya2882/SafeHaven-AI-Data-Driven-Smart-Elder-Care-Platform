# Reset index so we can manipulate the 'Cause' column

# Create a mapping function
def simplify_cause(name):
    name = name.lower()
    if "alzheimer" in name:
        return "Alzheimer"
    elif "dementia" in name:
        return "Dementia"
    elif "diabetes" in name:
        return "Diabetes"
    elif "covid" in name:
        return "COVID-19"
    elif "cardio" in name or "heart" in name:
        return "Cardio"
    elif "cancer" in name or "malignant neoplasm" in name:
        return "Cancer"
    elif "influenza" in name or "pneumonia" in name:
        return "Flu/Pneumonia"
    elif "suicide" in name or "self-harm" in name:
        return "Suicide"
    elif "accident" in name:
        return "Accidents"
    elif "parkinson" in name:
        return "Parkinson"
    elif "nephritis" in name or "kidney" in name:
        return "Kidney Disease"
    elif "stroke" in name:
        return "Stroke"
    elif "liver" in name:
        return "Liver Disease"
    elif "appendix" in name:
        return "Appendix"
    elif "unspecified" in name:
        return "Unspec. Infection"
    elif "respiratory" in name:
        return "Respiratory"
    elif "sepsis" in name:
        return "Sepsis"
    else:
        return name.split()[0].capitalize()  # Default to the first word

# Apply to dataframe

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# Define year columns
year_columns = [str(year) for year in range(2013, 2024)]

# Filter top 5 selected causes
top5_causes = ["Cancer", "Cardio", "Respiratory (All)", "Accidents", "Dementia"]
top5_df = df[df["Short_Cause"].isin(top5_causes)].copy()

# Convert year column names to integers for regression
x = np.array([int(year) for year in year_columns]).reshape(-1, 1)

# Plotting setup
plt.figure(figsize=(12, 7))

# Perform regression for each cause
for _, row in top5_df.iterrows():
    y = row[year_columns].values.astype(float)
    
    # Fit linear regression model
    model = LinearRegression()
    model.fit(x, y)
    y_pred = model.predict(x)

    # Plot actual values
    plt.plot(year_columns, y, marker='o', label=f"{row['Short_Cause']} Actual")

    # Plot regression line
    plt.plot(year_columns, y_pred, linestyle='--', label=f"{row['Short_Cause']} Trend")

# Final touches
plt.title("Linear Regression for Top 5 Causes of Death in Canada (2013–2023)")
plt.xlabel("Year")
plt.ylabel("Deaths per 100,000")
plt.legend()
plt.grid(True, linestyle='--', alpha=0.5)
plt.tight_layout()
plt.show()

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# Define year columns (existing) and future years to forecast
year_columns = [str(y) for y in range(2013, 2024)]
future_years = [2024, 2025, 2026]
all_years = year_columns + [str(y) for y in future_years]

# Prepare the plot
plt.figure(figsize=(12, 7))

# Loop through each row in the top5_df
for _, row in top5_df.iterrows():
    cause = row['Short_Cause']
    
    # Prepare features (X) and target (y)
    X = np.array([int(y) for y in year_columns]).reshape(-1, 1)
    y = row[year_columns].values.astype(float)
    
    # Fit linear regression model
    model = LinearRegression()
    model.fit(X, y)
    
    # Predict for existing + future years
    X_all = np.array([int(y) for y in all_years]).reshape(-1, 1)
    y_pred = model.predict(X_all)
    
    # Plot actual values
    plt.plot(year_columns, y, marker='o', label=f"{cause} Actual")
    
    # Plot predictions for all years
    plt.plot(all_years, y_pred, linestyle='--', label=f"{cause} Forecast")

# Final plot settings
plt.title("Linear Regression Forecast: Top 5 Causes of Death in Canada (2013–2026)")
plt.xlabel("Year")
plt.ylabel("Deaths per 100,000 Population")
plt.xticks(rotation=45)
plt.grid(True)
plt.legend(title="Cause")
plt.tight_layout()
plt.show()
